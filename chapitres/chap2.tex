Dans cette partie, toutes les structures sont des anneaux considérés comme
$\cL_{ring}$-structures donc plutôt que de noter $\cA_{ring}$ ou $\cA$, on
désignera par la même lettre $A$ à la fois le modèle et le domaine. 

Pour des raisons de convenances, on rappelle ici le langage et les axiomes pour

\begin{defi}[Anneaux, corps] \label{defi-ring}
Le langage naturel des anneaux est $\cL_{ring} = (+,\cdot,-,0,1)$.

On a les axiomes des groupes abéliens (dans le langage des groupes additifs
$\cL_{agp} =(+,-,0))$

\begin{enumerate}[label=\textbf{R\arabic*}]
    \item $\forall x,y,z [(x+y)+z = x+(y+z)]$,
    \item $\forall x [x+0 \wedge 0+x = x]$,
    \item $\forall x [x+(-x)=0 \wedge (-x)+x=0]$,
    \item $\forall x,y [x+y = y+x]$.
\end{enumerate}

Ensuite les axiomes de monoïde commutatif (dans le langage $\cL_{mon} = (\cdot,
1))$

\begin{enumerate}[label=\textbf{R\arabic*},start=5]
    \item $\forall x,y,z [(x\cdot y)\cdot z = x\cdot(y\cdot z)]$,
    \item $\forall x [x\cdot 1 \wedge 1\cdot x = x]$,
    \item $\forall x,y [x\cdot y = y\cdot x]$.
\end{enumerate}

enfin, la distributivité et l'\textit{unitarité}

\begin{enumerate}[label=\textbf{R\arabic*},start=8]
    \item $\forall x,y,z [x \cdot (y+z) = (x \cdot y) + (x \cdot z)]$.
    \item $0 \neq 1$,
\end{enumerate}

Ces huit phrases axiomatisent la théorie des anneaux commutatifs.

On rajoute ensuite l'axiome

\begin{enumerate}[label=\textbf{F\arabic*}]
    \item $\forall x [x = 0 \vee \exists y [x\cdot y = 1]]$.
\end{enumerate}

pour obtenir les axiomes de la théorie des corps.

\end{defi}

\begin{defi}[Anneau intègre]
Un anneau non trivial est \kw{intègre} si il  est sans diviseur de zéro, c'est à dire si
c'est un modèle pour la phrase \[ \forall x,y [x \cdot y = 0 \implies (x=0
\vee y=0) ]. \]
\end{defi}

\begin{defi}[Morphisme d'anneaux, noyau]
Un \kw{(homo)morphisme d'anneaux} est une application $f$ entre deux anneaux $A$ et
$B$ qui respecte la structure d'anneau, c'est à dire:
\begin{enumerate}[label= (\roman*)]
    \item $f(a+b) = f(a) + f(b), \forall a,b \in A$.
    \item $f(a \cdot b) = f(a) \cdot f(b), \forall a,b \in A$.
    \item $f(1_A) = 1_B$.
\end{enumerate}

Le noyau de $f$ est l'ensemble $\ker(f) = \{ a \in A \mid f(a) = 0 \} = f^{-1}({0})$.
\end{defi}

\begin{defi}[Idéal]
Un \kw{idéal} d'un anneau $A$ est un sous ensemble $I \subset A$ tel que:
\begin{enumerate}[label= (\roman*)]
    \item $I$ est un sous-groupe additif de $(A,+)$.
    \item $\forall a \in A, \forall x \in I, ax \in I$.
\end{enumerate}

L'\kw{idéal engendré} par un ensemble $J \subset A$ est l'intersection de tous
les idéaux de $A$ contenant $J$, on le note $(A)$

Si un idéal $I$ est tel que $I \subsetneq A$, on dit que l'idéal est \kw{propre}.
\end{defi}

\begin{prop}
Soit $f :A \implies B$ un morphisme d'anneaux et $J$ un idéal de $B$, alors
$I = f^{-1}(J)$ est un idéal de $A$. En particulier, $\ker(f)$ est un idéal de $A$.
\end{prop}

\begin{proof}
    On sait déjà que $I$ est une sous-groupe de $A$, ensuite pour $x\in I, a \in
    A$, on a $f(x) \in J$ donc \[ f(ax) = f(a)f(x) \in J \] car $J$ est un idéal
    et donc $ax \in J$ d'où le premier point.
    Comme $\{0\}$ est clairement un idéal, on a le second point. 
\end{proof}

\begin{defi}[Idéal premier, principal, maximal]
Un idéal $I$ d'une anneau $A$ est :
\begin{itemize}
    \item \kw{premier} si $\forall a,b \in A, ab \in I \implies (a \in I \vee b
    \in I)$
    \item \kw{principal} si $\exists a \in A, I = (a)$.
    \item \kw{maximal} si $I \neq A$ et pour tout idéal $J$ tel que $I \subset J
    \subset A$, on a $J = I \vee J = A$.
\end{itemize}

Si tout idéal de $A$ et principal, on dit que $A$ est un anneau \kw{principal}.
\end{defi}

\begin{defi}[Caractéristique,Sous-corps premier]
La \kw{caractéristique} d'un corps est, si il existe, le plus petit entier
strictement positif tel que $\underbrace{1+\cdots+1}_\text{p} = 0$, et $0$
sinon.
Le \kw{sous-corps premier} d'un corps $F$ est l'intersection de tous ses
sous-corps, c'est donc le sous-corps engendré par $1$.
Si $F$ est de caractéristique $p>0$, alors son sous-corps premier est
$\bZnZ{p}$, sinon c'est $\bQ$.
\end{defi}

\begin{prop}\label{prop-quotient-ideal-maximal}
Soit $I$ un idéal d'un anneau $A$, alors:
\begin{enumerate}[label= (\roman*)]
    \item $A/I$ est intègre si et seulement si $I$ est premier.
    \item $A/I$ est un corps si et seulement si $I$ est maximal.
\end{enumerate}
En particulier, tout idéal maximal est premier.
\end{prop}

\begin{proof}
\begin{enumerate}[label= (\roman*)]
    \item Supposons $A/I$ intègre, il n'est donc pas trivial donc $I \neq A$.
    Soit $a,b \in A$ tels que $ab \in I$, on a $\overline{ab} =
    \overline{a}\overline{b} = \overline{0}$ donc par intégrité, $\overline{a}=
    \overline{0} \vee \overline{b} = \overline{0}$ donc $a \in I \vee b \in I$
    donc $I$ est premier. Réciproquement, si $I$ est premier, $I \neq A$ donc
    $A/I$ n'est pas trivial. Soient $\overline{a},\overline{b} \in A/I$ tels que
    $\overline{a} \cdot \overline{b} = \overline{0}$, alors $\overline{ab} =
    \overline{0}$ donc $ab \in I$ donc comme $I$ est premier, $a \in I \vee b
    \in I$ c'est à dire $\overline{a} = 0 \vee \overline{b} = 0$ donc $A/I$ est intègre.
    \item Supposons que $A/I$ est un corps, comme précédemment on a $I \neq A$.
    Soit $J$ un idéal de $A$ tel que $I \subset J \subset A$, supposons $I \neq
    J$ et montrons qu'alors $J=A$. Soit $a \in J \backslash I$, on a $\overline{a} \neq
    \overline{0}$ donc il existe $\overline{b}\in A/I$ tel que
    $\overline{a}\overline{b} = 1$ et donc $\overline{1-ab} = \overline{0}$ donc
    $ab-1 \in I \subset J$ et donc $1 = (1-ab) + ab \in J$ ainsi, $J$ contient un
    élément inversible et donc $J=A$ et par suite, $I$ est maximal.
    Réciproquement, si $I$ est maximal, $I \neq A$ et $A/I$ est non trivial.
    Soit $\overline{a} \in (A/I) \backslash \{0\}$, soit $J = (a,I)$ l'idéal
    engendré par $a \not\in I$ et $I$. On a $I \subsetneq J$ donc comme $I$ est
    maximal, $J=A$ et particulier, $1 \in J$ donc il existe $b\in A, x\in I$
    tels que $ab+x = 1$ donc $\overline{a}\cdot\overline{b} = \overline{1}$ et
    $\overline{a}$ est inversible, ainsi $A/I$ est un corps.
\end{enumerate}

Comme un corps est un anneau intègre, par ce qui précède, tout idéal maximal est premier.

    
\end{proof}

\begin{theo}[Théorème de Krull]\label{theo-Krull}
Soit $I$ un idéal propre d'un anneau commutatif $A$, alors il existe un idéal
maximal de $A$ contenant $I$.
\end{theo}

\begin{defi}[Radical d'un idéal, idéal radiciel]\label{defi-ideal-radical}
Soit $I$ un idéal d'un anneau commutatif $A$, on appelle \kw{radical} de $I$ et on
note $\sqrt{I}$ l'ensemble des éléments de $A$ dont une puissance appartient à
$I$ : \[ \sqrt{I} = \{ a \in A \mid \exists n \in \bNe , a^n \in I \}.\]

Il est clair que $\sqrt{I}$ est un idéal de $A$, et qu'on a toujours $I \subset
\sqrt{I}$. Si on a l'égalité $I = V(I)$, on dit que l'idéal $I$ est \kw{radiciel}.
\end{defi}

Les corps algébriquement clos se manifestent comme des extensions de corps, on a

\begin{defi}[Extension de corps]
Soit $K$ un corps, une \kw{extension} de $K$ est un corps $L$ qui contient $K$
comme sous-corps. On le note $L/K$. Le corps $L$ muni de son produit et de la
loi externe $K \times L \implies L:(\lambda,x)\mapsto \lambda x$ est une
$K$-algèbre. La dimension de $L$ comme $K$-espace vectoriel est appelé
\kw{degré} de $L/K$ et noté $[L:K]$. Une \kw{sous-extension} de $L/K$ est un
sous-corps de $L$ contenant $K$.
\end{defi}

\begin{defi}[Sous-extension engendrée, extension simple]
Soit $L/K$ une extension de corps, et soit $E \subset L$, on note $K(E)$ l'intersection
des sous-corps contenant $K$ et $E$, c'est donc la plus petite sous-extension de
$L/K$ qui contient $E$. L'extension $K(E)/K$ est la \kw{sous-extension} de $L/K$
\kw{engendrée} par $E$. Si il existe $a \in L$ tel que $L=K(a)$, on
dit que l'extension $L/K$ est \kw{simple} ou \kw{monogène}.
\end{defi}

\begin{lemm}
Soit $K(a)/K$ une extension simple, il existe un unique morphisme (dit
d'évaluation) $ev_a : K[X] \implies K(a)$ tel que $\forall b \in K,
ev_a(b)=b$ et $ev_a(X) = a$.
\end{lemm}

\begin{defi}[Idéal annulateur]
Le noyau du morphisme d'évaluation $ev_a$ du lemme précédent est appelé

0 \forall b \in V \}. \]
\end{defi}  


\begin{prop}[\protect{$K[X]$} est principal]\label{prop-KX-principal}
    Soit $K$ un corps, alors $K[X]$ est principal et pour tout idéal $I = (P)$ non nul, les assertions suivantes sont équivalentes:
    \begin{enumerate}[label= (\roman*)]
        \item $P$ est irréductible.
        \item $I$ est un idéal premier.
        \item $I$ est un idéal maximal.
    \end{enumerate}
\end{prop}

\begin{defi}[Polynome minimal]
Soit $K(a)/K$ une extension simple, un polynôme $P$ qui engendre $I(a)$ est
appelé \kw{polynome minimal} de $a$.
\end{defi}

\begin{theo}[Existence d'un corps de rupture]\label{theo-corps-rupture}
    Soit $K$ un corps et $P \in K[X]$ un polynôme non constant.
    Il existe une extension $L$ de $K$ dans laquelle $P$ admet au moins une racine.
\end{theo}

\begin{proof}
    Supposons d'abord que $P$ est irréductible dans $K[X]$.
    Comme $K[X]$ est principal et que $P$ est irréductible, l'idéal $(P)$ est maximal (\cref{prop-KX-principal}).
    
    Donc d'après la \cref{prop-quotient-ideal-maximal} l'anneau quotient $L = K[X]/(P)$ est un corps.
    
    Montrons que $L$ est l'extension recherchée:
    \begin{enumerate}
        \item \textbf{$L$ contient $K$ :} Considérons la projection canonique $\pi : K[X] \to L$. La restriction de $\pi$ aux polynômes constants, $\phi : K \to L$ définie par $\phi(a) = \pi(a)$, est un morphisme d'anneaux. Comme $K$ est un corps, $\phi$ est injectif. On peut donc identifier $K$ à son image $\phi(K) \subset L$.
        \item \textbf{$P$ a une racine dans $L$ :} Soit $\alpha = \pi(X)$ la classe de l'indéterminée $X$ dans $L$. Évaluons le polynôme $P = \sum a_i X^i$ en $\alpha$ :
        \[ P(\alpha) = \sum_{i} a_i (\pi(X))^i = \sum_{i} \pi(a_i X^i) = \pi\left( \sum_{i} a_i X^i \right) = \pi(P) \]
        Or, par construction, $P \in (P)$, donc $\pi(P) = 0$ dans $L$.
        Ainsi, $P(\alpha) = 0$ et $\alpha$ est une racine de $P$ dans $L$.
    \end{enumerate}
    
    Et si $P$ n'est pas irréductible, il suffit d'appliquer ce qui
    précède sur un de ses facteurs irréductibles.
\end{proof}


\begin{prop}[Classification des extensions simples]
Soit $K(a)/K$ une extension simple. De ce qui précède, on déduit que l'extension
est entièrement déterminée, à isomorphisme près, par le choix d'un polynôme
minimal de $a$ qui est irréductible. Donc classifier les extensions simples
revient à classifier les idéaux premiers de $K[X]$.
\end{prop}

\begin{exem}
Prenons $K=\bQ$ et $a = \sqrt{2} \in \bQ(\sqrt{2})$. Le polynôme minimal de
$\sqrt{2}$ est $X^2-2$. 
Prenons maintenant $a = \pi$, comme $\pi$ est transcendant, son polynôme minimal
est le polynôme nul et $I(\pi) = \{0\}$ donc $ev_{\pi}: \bQ[X] \implies
\bQ(\pi)$ est injectif et par suite, $\bQ(r) \simeq \bQ(X)$. 
\end{exem}

\begin{defi}[Extension finie]
Soit $L/K$ une extension de corps, si il existe $\overline{a} =
(a_1,\ldots,a_n) \in L^n$ tel que $L = K(\overline{a})$, on dit que l'extension est \kw{finie}.
\end{defi}

\begin{prop}[Classification des extensions finies]
Soit $K(a_1,\ldots,a_n)/K$ une extension finie. Comme dans le cas simple, on a
$K(a_1,\ldots,a_n) \simeq K[X_1,\ldots,X_n]/I(a_1,\ldots,a_n)$ donc pour classifier
les extensions finies, il suffit de classifier les idéaux premiers de
$K[X_1,\ldots,X_n]$ pour tout $n \in \bNe$.
\end{prop}

\begin{theo}[Base de Hilbert]\label{theo-base-Hilbert}
Soit $K$ un corps et $n\in \bNe$ alors tout idéal de $K[X_1, \ldots, X_n ]$ est
engendré par un nombre fini de polynômes.
\end{theo}

\begin{defi}[Extension algébrique]
Soit $L/K$ une extension de corps, si tout élément de $F$ est algébrique sur
$K$, on dit que $L/K$ est une \kw{extension algébrique}.
\end{defi}

\begin{prop}
Si $L/K$ est une extension algébrique finie, alors c'est une extension simple.
\end{prop}

\begin{lemm}
Si $L/K$ et  $K/F$ sont des extensions algébriques, alors $L/F$ également.
\end{lemm}


\section{Clôture algébrique d'un corps}

\begin{defi}[Corps algébriquement clos,clôture algébrique]
On dit qu'un corps $K$ est \kw{algébriquement clos} si tout polynôme non constant de
$K[X]$ admet une racine dans $K$. On dit que $L$ est une \kw{clôture algébrique}
de $K$ si $L/K$ est une extension algébrique et $L$ est algébriquement clos.
\end{defi}

On va avoir besoin des lemmes suivants:

\begin{lemm}\label{lemm-extension-racine}
    Pour tout corps $K$, il existe une extension $K^{*}/K$ telle que tout polynôme non constant $P \in K[X]$ admette au moins une racine dans $K^*$.
\end{lemm}

\begin{proof}
    Considérons le langage des anneaux auquel on ajoute les constantes de $K$ et
   une constante pour chaque polynôme unitaire non constant $P \in K[X]$, \[ \cL_K =
   \cL_{ring} \cup \{c_x \mid x \in K\}  \cup \{c_P \mid P\in K[X] \text{ unitaire
   non constant}\}\]

    Soit $T$ la théorie constituée de:
    \begin{enumerate}
        \item $\Diag(K)$ (pour assurer que le modèle sera une extension de $K$).
        \item L'axiome $P(c_P) = 0$ pour chaque $P \in K[X]$.
    \end{enumerate}
    
    Soit $T_0$ un sous-ensemble fini de $T$. Il ne fait intervenir qu'un nombre
    fini de polynômes $P_1, \ldots , P_m$. D'après le \cref{theo-corps-rupture} il
    existe une extension finie $L$ de $K$ contenant une racine $r_i$ pour chaque
    $P_i$.
    En interprétant chaque constante  $c_{P_i}$ par la racine $r_i$,  on a $L
    \models T_0$.
    
    Ainsi, tout sous-ensemble fini de $T$ est satisfiable, par le théorème
    de compacité (\cref{theo-compacite}) la théorie $T$ est satisfiable. Soit
    $K^{*} \models T$, comme $K^{*} \models \Diag(K)$, $K \subset K^{*}$ et par
    construction, tout polynôme non constant de $K[X]$ y admet une racine d'où
    le résultat.
\end{proof}

\begin{lemm}[Union d'une chaîne de corps]\label{lemm-chaine-corps}
    Soit $(K_n)_{n \in \mathbb{N}}$ une suite de corps telle que pour tout $n$, $K_n$ est un sous-corps de $K_{n+1}$ (on parle d'une \textbf{chaîne croissante}).
    Alors l'ensemble $K_\infty = \bigcup_{n \in \mathbb{N}} K_n$, muni des opérations induites, est un corps.
\end{lemm}

\begin{proof}
    Soient $x, y \in K_\infty$. Par définition de la réunion, il existe des indices $i$ et $j$ tels que $x \in K_i$ et $y \in K_j$. Posons $N = \max(i, j)$. Par croissance de la chaîne, $K_i \subseteq K_N$ et $K_j \subseteq K_N$, donc $x$ et $y$ appartiennent tous deux au corps $K_N$.
    
    Les opérations $x+y$, $x \cdot y$, $-x$ et $x^{-1}$ (si $x \ne 0$) sont donc bien définies dans $K_N$ et par suite dans $K_\infty$. Les axiomes des corps (associativité, distributivité, etc.) étant vérifiés dans chaque $K_n$, ils le sont dans l'union.
\end{proof}

On peut maintenant montrer le résultat suivant:

\begin{prop}\label{prop-existence-cloture-algebrique}
Tout corps admet une clôture algébrique.
\end{prop}

\begin{proof}
    Soit $K$ un corps,
    Définissons par récurrence une chaîne de corps $(K_n)_{n \in \mathbb{N}}$ :
    \begin{itemize}
        \item $K_0 = K$.
        \item $K_{n+1}$ est l'extension de $K_n$ obtenue par le \cref{lemm-extension-racine} (telle que tout polynôme de $K_n[X]$ ait une racine dans $K_{n+1}$).
    \end{itemize}
    
    Soit $L = \bigcup_{n \in \mathbb{N}} K_n$. D'après le \cref{lemm-chaine-corps}, $L$ est un corps.
    Ensuite, soit $P \in L[X]$ un polynôme non constant. 
    Comme $P$ possède un nombre fini de coefficients, il existe un rang $n$ tel que tous les coefficients de $P$ appartiennent à $K_n$, c'est-à-dire $P \in K_n[X]$.
    Par construction, $P$ admet une racine dans $K_{n+1}$ et donc dans $L$ donc
    $L$ est algébriquement clos.
    En posant $\overline{K}=  \{ x \in L \mid x \text{ algébrique sur} K \}$, il
    est immédiat que $\overline{K}$ que est une clôture algébrique de $K$.
\end{proof}

\begin{theo}[Théorème de Steinitz]
Tout corps $K$ admet une clôture algébrique, unique à isomorphisme près. On la

\begin{defi}[Indépendance algébrique, base de transcendance]
Soit $L/K$ une extension de corps, un sous-ensemble $E \subset L$ est
\kw{algébriquement indépendant} sur $K$ si pour tout $n \in \bNe$ et $P \in
K[X_1, \ldots, X_n]$ non nul, si $e_1, \ldots, e_n \in E$ sont distincts, alors
$P(e_1,\ldots, e_n) \neq 0$.
Si $E$ est algébriquement indépendant sur $K$ et que $L$ est une extension algébrique de
$K(E)$, alors on dit que $E$ est une base de transcendance de $L/K$.
\end{defi}

\begin{prop}
Toute extension de corps admet une base de transcendance et toutes les bases de
transcendance d'une extension ont un même cardinal appelé le \kw{degré de
transcendance} de l'extension.
\end{prop}

\begin{prop}\label{prop-alg-clos-isom}
Deux corps algébriquement clos sont isomorphes si et seulement si ils ont même
caractéristique et même degré de transcendance 
\end{prop}

\begin{theo}[Théorème de structure des corps finis]\label{theo-structure-corps-fini}
Soit $p \in \bP, k \in \bNe$. 
L'ensemble des racines du polynôme $P(X) = X^{p^k} - X$ dans $\Fpalg$
est un sous-corps de $\Fpalg$ noté $\bF_{p^k}$ et c'est l'unique corps
de cardinal $p^k$ à isomorphisme près et de plus, $\Fpalg = \bigcup_{k
\ge 1} \bF_{p^k}$.
\end{theo}

\section{Théorie des corps algébriquement clos}

On a maintenant tous les outils algébriques nécessaires et pouvons revenir à la
théorie des modèles. On commence par définir la théorie des corps
algébriquement clos (ACF). Puis on montre successivement qu'elle admet
l'élimination des quantificateurs, la complétude ainsi que le principe de
Lefschetz qui nous seront indispensables pour la suite. On étudie ensuite les
ensembles définissables de la théorie et leur lien avec les variétés algébriques.

\begin{defi}[Théorie des corps algébriquement clos]
    On se place dans le langage des anneaux $\cL_{ring}$ avec les axiomes des
    corps (\cref{defi-ring}).\\
    On axiomatise les corps algébriquement clos en rajoutant, pour
    chaque $n \in \bNe$, l'axiome \[ \forall a_0, \ldots, a_{n-1}, \exists x
    [x^n+a_{n-1}\cdot x^{n-1} + \cdots + a_1 \cdot x + a_0 = 0].\]

    On note $ACF$ l'ensemble des axiomes des corps algébriquement clos. \\

    On axiomatise la caractéristique $p \in \bP$ en notant $\chi_p : \underbrace{1 +
    \cdots 1 = 0}_\text{p} $.

    On pose alors \[ \ACF_p := \Th(ACF \cup \chi_p) \] la théorie des corps
    algébriquement clos de caractéristique $p$ et \[ \ACF_0 := \Th(ACF \cup \{\neg
    \chi_p \mid p \in \bP\}) \] la théorie des corps algébriquement clos de
    caractéristique nulle. On notera $\ACF_n$ la théorie des corps
    algébriquement clos de caractéristique $n$ pour $n \in \bP \cup \{0\}$. 
\end{defi}

\begin{defi}[Elimination des quantificateurs]\label{def-quant-elim}
On dit qu'une théorie $T$ a l'\kw{élimination des quantificateurs} si pour toute
formule $\varphi$, il existe une formule sans-quantificateurs $\theta$ telle que
$T \models \varphi \leftrightarrow \theta$. \\
Plus précisément, une $\cL$-théorie
$T$ a l'\kw{élimination des quantificateurs} dans $\cL$ si pour toute
$\cL$-formule $\varphi(x_1, \ldots, x_n)$, il existe une $\cL$-formule
$\theta(x_1, \ldots, x_m)$ sans quantificateurs, où $n \le m$ telle que \[ T
\models \forall x_1 \cdots \forall x_m ( \varphi(x_1, \ldots, x_n)
\leftrightarrow \theta(x_1, \ldots, x_m)) \] 
\end{defi}

\begin{prop}[Complétude]\label{prop-acf-complet}
    La théorie $\ACF_n$ est complète.
\end{prop}

\begin{proof}
On montre le résultat pour $\ACF_p$, la preuve pour $\ACF_0$ est identique en
remplaçant $\bF_p$ par $\bQ$. \\
Soient $K,L$ tels que $K,L \models \ACF_p$ et $\varphi$ une phrase dans le
langage des anneaux. Par l'élimination des quantificateurs, il existe une phrase
$\theta$ telle que $\ACF_n \models \varphi \leftrightarrow \theta$. Par le
\cref{lemm-sous-struct-sans-quant}, $K \models \theta \models \leftrightarrow
\bF_p \models \theta \iff L \models \theta$ et par donc $K \models \varphi
\leftrightarrow K \models \theta \leftrightarrow L \models \theta
\leftrightarrow L \models \varphi$ et donc $K$ et $L$ sont équivalentes et
$\ACF_p$ est complète.
\end{proof}

\begin{coro}[Principe de Lefschetz]\label{coro-Lefschetz}
    Soit $\varphi$ une phrase du langage des anneaux. On a les
    équivalences suivantes

    \begin{enumerate}[label= (\roman*)]
        \item $\varphi$ est vraie sur  $\bC$.
        \item Il existe un corps algébriquement clos de caractéristique $0$ sur
        lequel $\varphi$ est vraie.
        \item $\varphi$ est vraie sur tout corps algébriquement clos de
        caractéristique $0$.
        \item Il existe $M>0$ tel que pour tout premier $p>M$, il existe un
        corps algébriquement clos de caractéristique $p$ sur lequel $\varphi$
        est vraie.
        \item Il existe des premiers $p$ arbitrairement grands et des corps
        algébriquement clos de caractéristique $p$ sur lesquels $\varphi$ est
        vraie.
    \end{enumerate}

\end{coro}

\begin{proof}
On montre que $1 \implies 2 \implies 3 \implies 1$ puis $3 \implies
4 \implies 5 \implies 2$. \\
Il est clair que $1 \implies 2$ et  $3 \implies 1$, $2 \implies 3$
est une conséquence immédiate de la complétude (\cref{prop-acf-complet}).\\
$(3 \implies 4)$: On a $\ACF_0 \models \varphi$ donc par compacité
(\cref{theo-compacite}), il existe $\Sigma \subset \ACF_0$ fini tel que $\Sigma
\models \varphi$. Comme $\Sigma$ est fini, l'ensemble $E= \{ p \in \bP \mid \neg
\chi_p \in \Sigma$ aussi. En posant $M = \max(E)$ (et $M=0$ si $E = \emptyset)$,
pour $p > M$, on a $ \ACF_p \models \Sigma$ et donc $\ACF_p \models \varphi$. \\
$4 \implies 5$ est claire et enfin, \\
$(5 \implies 2)$: Supposons que $\ACF_0 \not\models \varphi$,
par la \cref{prop-acf-complet} $\ACF_0$ est complète ainsi $\ACF_0 \models \neg
\varphi$ donc par ce qui précède, $\ACF_p \models \neg\varphi$ pour $p$
suffisamment grand ce qui contredit $4$ d'où le résultat.


\end{proof}

\begin{defi}[Variété affine]
Soit $S \subset K[\overline{x}]$, on note $V(S)$ le sous ensemble de $K^n$
formé des zéros communs à tous les éléments de $S$. C'est à dire: 
\[ V(S) = \{ \overline{a} \in K^n \mid P(\overline{a} ) = 0,  \forall P \in S \} .\]
Les sous-ensembles de $K^n$ de ce type sont les sous-variétés algébrique affine de
$K^n$.
Dans la suite on utilisera simplement \kw{variété} pour les qualifier.
\end{defi}

\begin{defi}[Idéal d'une variété]
Soit $V \subset K^n$, on appelle idéal de $V$, et on note $I(V)$ l'ensemble des
polynômes nuls sur $V$. C'est à dire:
\[ I(V) = \{ P \in  K[\overline{x}] \mid P(\overline{a}) = 0, \forall \overline{a} \in V \}.\]
\end{defi}
