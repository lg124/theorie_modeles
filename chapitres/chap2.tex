
\chapter{Complétude d'ACF et conséquences}

\section{Intermède algébrique I}

On rappelle ici quelques définitions et propriétés qui nous seront utiles
par la suite. On pourra trouver les démonstrations
dans n'importe quel livre d'algèbre, par exemple dans~\cite{Berhuy_2025}.


\begin{defi}[Extension de corps]
Soit $K$ un corps, une \kw{extension} de $K$ est un corps $L$ qui contient $K$
comme sous-corps. On le note $L/K$. Le corps $L$ muni de son produit et de la
loi externe $K \times L \implies L:(\lambda,x)\mapsto \lambda x$ est une
$K$-algèbre. La dimension de $L$ comme $K$-espace vectoriel est appelé
\kw{degré} de $L/K$ et noté $[L:K]$, si il est fini, l'extension est dite
\kw{finie}.Une \kw{sous-extension} de $L/K$ est un
sous-corps de $L$ contenant $K$.
\end{defi}

\begin{defi}[Sous-extension engendrée, extension simple]
Soit $L/K$ une extension de corps, et soit $E \subset L$, on note $K(E)$ l'intersection
des sous-corps contenant $K$ et $E$, c'est donc la plus petite sous-extension de
$L/K$ qui contient $E$. L'extension $K(E)/K$ est la \kw{sous-extension} de $L/K$
\kw{engendrée} par $E$. Si il existe $a \in L$ tel que $L=K(a)$, on
dit que l'extension $L/K$ est \kw{simple} ou \kw{monogène}.
\end{defi}

\begin{lemm}
Soit $K(a)/K$ une extension simple, il existe un unique morphisme (dit
d'évaluation) $ev_a : K[X] \implies K(a)$ tel que $\forall b \in K,
ev_a(b)=b$ et $ev_a(X) = a$.
\end{lemm}


%\todo[inline]{redondant avec idéal d'une variété? a del?}
\begin{defi}[Idéal annulateur]
Le noyau du morphisme d'évaluation $ev_a$ du lemme précédent est appelé
\kw{idéal anulteur} et noté $I(a)$, on a donc \[ I(a) := \{ f \in K[x] \mid f(a) =
0 \}. \]
Plus généralement, si $V\subset K$ est un ensemble (algébrique), l'\kw{idéal
annulateur} de $V$ est l'ensemble \[ I(V) := \{ f \in K[x] \mid f(V) =
0 \forall b \in V \}. \]
\end{defi}  


\begin{prop}[\protect{$K[X]$} est principal]\label{prop-KX-principal}
    Soit $K$ un corps, alors $K[X]$ est principal et pour tout idéal $I = (P)$ non nul, les assertions suivantes sont équivalentes:
    \begin{enumerate}[label= (\roman*)]
        \item $P$ est irréductible.
        \item $I$ est un idéal premier.
        \item $I$ est un idéal maximal.
    \end{enumerate}
\end{prop}

\begin{defi}[Polynome minimal]
Soit $K(a)/K$ une extension simple, un polynôme $P$ qui engendre $I(a)$ est
appelé \kw{polynome minimal} de $a$.
\end{defi}

\begin{theo}[Existence d'un corps de rupture]\label{theo-corps-rupture}
    Soit $K$ un corps et $P \in K[X]$ un polynôme non constant.
    Il existe une extension $L$ de $K$ dans laquelle $P$ admet au moins une racine.
\end{theo}

\begin{defi}[Extension algébrique]
Soit $L/K$ une extension de corps, si tout élément de $L$ est algébrique sur
$K$, on dit que $L/K$ est une \kw{extension algébrique}.
\end{defi}

\section{Cloture algébrique d'un corps}

\begin{defi}[Corps algébriquement clos,clôture algébrique]
On dit qu'un corps $K$ est \kw{algébriquement clos} si tout polynôme non constant de
$K[X]$ admet une racine dans $K$. On dit que $L$ est une \kw{clôture algébrique}
de $K$ si $L/K$ est une extension algébrique et $L$ est algébriquement clos.
\end{defi}

On va avoir besoin des résultats suivants:

\begin{defi}[Diagramme d'une structure]\label{defi-diagramme}
    Soit $\cM$ une $\cL$-structure et $\cL_{\cM}$ le language où on ajoute à $\cL$
    des constantes $m$ pour chaque élément de $\cM$. Le \kw{diagramme} (atomique) de
    $\cM$, noté  $\Diag(\cM)$ est l'ensemble $\{ \varphi(m_1, \ldots, m_n) :
    \varphi$  est une $\cL$-formule atomique ou la négation d'une $\cL$-formule
    atomique et $\cM \models \varphi(m_1, \ldots, m_n)  $\}.

\end{defi}

\begin{lemm}\label{lemm-extension-racine}
    Pour tout corps $K$, il existe une extension $L/K$ telle que tout polynôme non constant $P \in K[X]$ admette au moins une racine dans $L$.
\end{lemm}

\begin{proof}
    Considérons le langage des anneaux auquel on ajoute les constantes de $K$ et
   une constante pour chaque polynôme unitaire non constant $P \in K[X]$, \[ \cL_K =
   \cL_{ring} \cup \{c_x \mid x \in K\}  \cup \{c_P \mid P\in K[X] \text{ unitaire
   non constant}\}\]

    Soit $T$ la théorie constituée de:
    \begin{enumerate}
        \item $\Diag(K)$ (pour assurer que le modèle sera une extension de $K$).
        \item L'axiome $P(c_P) = 0$ pour chaque $P \in K[X]$.
    \end{enumerate}
    
    Soit $T_0$ un sous-ensemble fini de $T$. Il ne fait intervenir qu'un nombre
    fini de polynômes $P_1, \ldots , P_m$. D'après le \cref{theo-corps-rupture} il
    existe une extension finie $L$ de $K$ contenant une racine $r_i$ pour chaque
    $P_i$.
    En interprétant chaque constante  $c_{P_i}$ par la racine $r_i$,  on a $L
    \models T_0$.
    
    Ainsi, tout sous-ensemble fini de $T$ est satisfiable, par le théorème
    de compacité (\cref{theo-compacite}) la théorie $T$ est satisfiable. Soit
    $L \models T$, comme $L \models \Diag(K)$, $K \subset L$ et par
    construction, tout polynôme non constant de $K[X]$ y admet une racine d'où
    le résultat.
\end{proof}

\begin{lemm}[Union d'une chaîne de corps]\label{lemm-chaine-corps}
    Soit $(K_n)_{n \in \mathbb{N}}$ une suite de corps telle que pour tout $n$, $K_n$ est un sous-corps de $K_{n+1}$ (on parle d'une \textbf{chaîne croissante}).
    Alors l'ensemble $K_\infty = \bigcup_{n \in \mathbb{N}} K_n$, muni des opérations induites, est un corps.
\end{lemm}

\begin{proof}
    Soient $x, y \in K_\infty$. Par définition de la réunion, il existe des indices $i$ et $j$ tels que $x \in K_i$ et $y \in K_j$. Posons $N = \max(i, j)$. Par croissance de la chaîne, $K_i \subseteq K_N$ et $K_j \subseteq K_N$, donc $x$ et $y$ appartiennent tous deux au corps $K_N$.
    
    Les opérations $x+y$, $x \cdot y$, $-x$ et $x^{-1}$ (si $x \ne 0$) sont donc bien définies dans $K_N$ et par suite dans $K_\infty$. Les axiomes des corps (associativité, distributivité, etc.) étant vérifiés dans chaque $K_n$, ils le sont dans l'union.
\end{proof}

On peut maintenant montrer le résultat suivant:

\begin{prop}\label{prop-existence-cloture-algebrique}
Tout corps admet une clôture algébrique.
\end{prop}

\begin{proof}
    Soit $K$ un corps,
    Définissons par récurrence une chaîne de corps $(K_n)_{n \in \mathbb{N}}$ :
    \begin{itemize}
        \item $K_0 = K$.
        \item $K_{n+1}$ est l'extension de $K_n$ obtenue par le \cref{lemm-extension-racine} (telle que tout polynôme de $K_n[X]$ ait une racine dans $K_{n+1}$).
    \end{itemize}
    
    Soit $L = \bigcup_{n \in \mathbb{N}} K_n$. D'après le \cref{lemm-chaine-corps}, $L$ est un corps.
    Ensuite, soit $P \in L[X]$ un polynôme non constant. 
    Comme $P$ possède un nombre fini de coefficients, il existe un rang $n$ tel que tous les coefficients de $P$ appartiennent à $K_n$, c'est-à-dire $P \in K_n[X]$.
    Par construction, $P$ admet une racine dans $K_{n+1}$ et donc dans $L$ donc
    $L$ est algébriquement clos.
    En posant $\overline{K}=  \{ x \in L \mid x \text{ algébrique sur} K \}$, il
    est immédiat que $\overline{K}$ que est une clôture algébrique de $K$.
\end{proof}


\begin{defi}[Indépendance algébrique, base de transcendance]
Soit $L/K$ une extension de corps, un sous-ensemble $E \subset L$ est
\kw{algébriquement indépendant} sur $K$ si pour tout $n \in \bNe$ et $P \in
K[X_1, \ldots, X_n]$ non nul, si $e_1, \ldots, e_n \in E$ sont distincts, alors
$P(e_1,\ldots, e_n) \neq 0$.
Si $E$ est algébriquement indépendant sur $K$ et que $L$ est une extension algébrique de
$K(E)$, alors on dit que $E$ est une base de transcendance de $L/K$.
\end{defi}

\begin{prop}
Toute extension de corps admet une base de transcendance et toutes les bases de
transcendance d'une extension ont un même cardinal appelé le \kw{degré de
transcendance} de l'extension.
\end{prop}

\begin{theo}[Théorème de structure des corps finis]\label{theo-structure-corps-fini}
Soit $p \in \bP, k \in \bNe$. 
L'ensemble des racines du polynôme $P(X) = X^{p^k} - X$ dans $\Fpalg$
est un sous-corps de $\Fpalg$ noté $\bF_{p^k}$ et c'est l'unique corps
de cardinal $p^k$ à isomorphisme près et de plus, $\Fpalg = \bigcup_{k
\ge 1} \bF_{p^k}$.
\end{theo}

\begin{theo}[Théorème de Steinitz]\label{theo-Steinitz}
Tout corps $K$ admet une clôture algébrique, unique à isomorphisme près. On la notera $\overline{K}$.
De plus, deux corps algébriquement clos sont isomorphes si et seulement s'ils ont même caractéristique et même degré de transcendance.
\end{theo}


\section{Complétude, categoricité et test de Vaught}

\begin{defi}[Théorie complète]
Une $\mathcal{L}$-théorie $T$ est dite complète si pour toute $\mathcal{L}$-phrase $\phi$, on a soit $T \models \phi$, soit $T \models \neg\phi$.
\end{defi}

\begin{defi}[Categoricité]
Soient $\kappa$ un cardinal infini et $T$ une $\cL$-théorie satisfaite par des modèles de cardinalité $\kappa$.
On dit que $T$ est $\kappa$-catégorique si les modèles de $T$ de cardinalité $\kappa$ sont deux à deux isomorphes.
\end{defi}

\noindent Nous donnons un premier exemple de théorie $\kappa$-catégorique: la théorie au langage vide forçant ses modèles à être infini.

\begin{prop}
Soit $\cL = \emptyset$. Pour $n \in \mathbb{N^*}$, on définit la $\cL$-phrase
\begin{align*}
\phi_n = \exists x_1\dots\exists x_n(\bigwedge_{1\leq i < j \leq n} x_i \neq x_j)
\end{align*}
On définit la $\cL$-théorie $T_{\infty} = \{\phi_n,\text{ }n \in \mathbb{N^*}\}$.\\
Alors pour tout cardinal $\kappa$, $T_{\infty}$ est $\kappa$-catégorique.
\end{prop}

\begin{proof}
Soit $\kappa$ un cardinal infini. Soient $\cM$ et $\cN$ deux modèles de $T_{\infty}$ tel que $\lvert \cM \rvert = \lvert \cN \rvert = \kappa$.
Puisque $\cL = \emptyset$, nos modèles sont simplement des ensembles. Or par un argument ensembliste, $M$ et $N$ étant de même cardinalité, il existe $\phi: M \longrightarrow N$ une bijection.
$\phi$ étant une bijection entre structures vides, il n'y a aucun symbole à preserver et c'est un $\cL$-isomorphisme.
\end{proof}

\noindent Nous présentons le résultat suivant comme fait algébrique. Il est le pilier pour distinguer des modèles de $\ACF_p$ de même cardinalité via des invariants.

\begin{coro}\label{acfcategorique}
Soit $p \in \{0\} \cup \mathcal{P}$, alors $ACF_p$ est $\kappa$-catégorique pour tout cardinal $\kappa$ indénombrable.
\end{coro}

\begin{proof}
Il est clair que deux modèles de $ACF_p$ sont tous deux de caractéristique $p$. De plus un corps algébriquement clos de degré de transcendance $\lambda$ a
une cardinalité égale à $\max(\lambda, \aleph_0)$. Ainsi soit $\kappa > \aleph_0$ et un corps algébriquement clos de cardinalité $\kappa$, alors le 
degré de transcendance est aussi égal à $\kappa$. Ainsi deux modèles de $ACF_p$
de même cardinalité indénombrable sont isomorphes par théorème de Steinitz
(\cref{theo-Steinitz}).
\end{proof}

\noindent Le Théorème de compacité assure qu'une théorie reste satisfiable tant que ses sous-ensembles finis le sont.
Pour une théorie $T$ finiment satisfiable et à modèles infinis, on peut ajouter une infinité de symboles de constante et imposer qu'ils soient deux à deux distincts.
Ceci crée une nouvelle théorie, toujours finiment satisfiables (toute partie finie de la nouvelle théorie ne demandent qu'un nombre fini de constantes, lesquelles peuvent 
être choisies par des éléments distincts d'un modèle infini de $T$), et imposant par compacité l'existence de modèles à cardinalité aussi grande que l'on souhaite.
Ce raisonnement fonde le résultat suivant :


\begin{theo}[Löwenheim-Skolem ascendant]
Soit $T$ une $\cL$-théorie satisfaite par des modèles infinis. Si $\kappa$ est un cardinal infini tel que $\kappa > \lvert \cL \rvert$, alors
il existe un modèle de $T$ de cardinalité $\kappa$.
\end{theo}

\begin{theo}[Test de Vaught]
Soit $T$ une $\cL$-théorie satisfiable, sans modèles finis, et $\kappa$-catégorique pour un certain cardinal infini $\kappa > \lvert \cL \rvert$.
Alors $T$ est complète.
\end{theo}

\begin{proof}
Supposons par l'absurde que $T$ n'est pas complète. Il existe alors une $\cL$-phrase $\phi$ telle que $T \not\models \phi$ et $T \not\models \neg\phi$.\\
\textbf{Fait :} $T \not\models \phi$ si et seulement si $T \cup \{\neg\phi\}$ est satisfiable. En effet, si $T \models \phi$, alors $\neg\phi$ ne se satisfait dans aucun modèle de $T$,
donc $T \cup \{\neg\phi\}$ est insatisfiable car un tel modèle devrait satisfaire à la fois $T$ et à la fois $\neg\phi$. Réciproquement, si $T \cup \{\neg\phi\}$ est insatisfiable,
cela signifie qu'un modèle de $T$ ne peut pas aussi satisfaire à $\neg\phi$, donc un tel modèle satisfait nécessairement $\phi$ par définition de la satisfaisabilité d'une formule, d'où $T \models \phi$. \\
Ainsi $T_0 = T \cup \{\phi\}$ et $T_1 = T \cup \{\neg\phi\}$ sont satisfiables.
Par hypothèse $T$ n'a pas de modèles finis : ceci impose que les modèles de $T_0$ et $T_1$ soient infinis. Par Théorème de Löwenheim-Skolem ascendant, on a $\cM_0$ et $\cM_1$
tout deux de cardinalité $\kappa > \lvert \cL \rvert$ tel que $\cM_0 \models T_0$ et $\cM_1 \models T_1$. Or ces deux modèles ne peuvent être élémentairement équivalent car ils divergent sur $\phi$.
Par théorème~\ref{isomorphisme}, ces modèles ne sont pas isomorphes. Ceci contradit la $\kappa$-catégoricité de $T$.
\end{proof}

\begin{prop} \label{prop-acf-complet}
Soit $p \in \{0\} \cup \mathcal{P}$, alors $ACF_p$ est complète.
\end{prop}

\begin{proof}
Immédiat par corollaire~\ref{acfcategorique} et Test de Vaught.
\end{proof}

\section{Principe de Lefschetz}

La complétude d'ACF est un résultat puissant qui va nous permettre de
généraliser facilement des résultats démontrés sur des cas particuliers. En
effet, comme toutes les vérités (du premier ordre) sont partagées par tous les
modèles d'une même caractéristique, on va pouvoir transférer un théorème
démontré sur un corps spécifique (par exemple $\bC$ sur lequel on dispose de
nombreux outils analytiques ou topologiques) vers tout autre corps de même caractéristique.
Le résultat suivant généralise ce principe en montrant qu'on peut aussi faire le
lien entre la caractéristique nulle et la caractéristique positive.

\begin{coro}[Principe de Lefschetz]\label{coro-Lefschetz}
    Soit $\varphi$ une phrase du langage des anneaux. On a les
    équivalences suivantes

    \begin{enumerate}[label= (\roman*)]
        \item $\varphi$ est vraie sur  $\bC$.
        \item Il existe un corps algébriquement clos de caractéristique $0$ sur
        lequel $\varphi$ est vraie.
        \item $\varphi$ est vraie sur tout corps algébriquement clos de
        caractéristique $0$.
        \item Il existe $M>0$ tel que pour tout premier $p>M$, il existe un
        corps algébriquement clos de caractéristique $p$ sur lequel $\varphi$
        est vraie.
        \item Il existe des premiers $p$ arbitrairement grands et des corps
        algébriquement clos de caractéristique $p$ sur lesquels $\varphi$ est
        vraie.
    \end{enumerate}

\end{coro}

\begin{proof}
On montre que $1 \implies 2 \implies 3 \implies 1$ puis $3 \implies
4 \implies 5 \implies 2$. \\
Il est clair que $1 \implies 2$ et  $3 \implies 1$, $2 \implies 3$
est une conséquence immédiate de la complétude (\cref{prop-acf-complet}).\\
$(3 \implies 4)$: On a $\ACF_0 \models \varphi$ donc par compacité
(\cref{theo-compacite}), il existe $\Sigma \subset \ACF_0$ fini tel que $\Sigma
\models \varphi$. Comme $\Sigma$ est fini, l'ensemble $E= \{ p \in \bP \mid \neg
\chi_p \in \Sigma \}$ aussi. En posant $M = \max(E)$ (et $M=0$ si $E = \emptyset)$,
pour $p > M$, on a $ \ACF_p \models \Sigma$ et donc $\ACF_p \models \varphi$. \\
$4 \implies 5$ est claire et enfin, \\
$(5 \implies 2)$: Supposons que $\ACF_0 \not\models \varphi$,
par la \cref{prop-acf-complet} $\ACF_0$ est complète ainsi $\ACF_0 \models \neg
\varphi$ donc par ce qui précède, $\ACF_p \models \neg\varphi$ pour $p$
suffisamment grand ce qui contredit $4$ d'où le résultat.
\end{proof}

\section{Théorème d'Ax-Grothendieck}

Dans ce qui précède, nous avons introduit la théorie des modèles et établi
certains résultats, notamment sur les corps algébriquement clos. Il est
maintenant temps d'appliquer ces résultats et de démontrer que la puissance de
la théorie des modèles permet de résoudre des problèmes concrets provenant
d'autres branches des mathématiques.

La première application que nous étudions, le \kw{théorème
d'Ax-Grothendieck}, illustre la puissance du principe de transfert en
transformant un problème difficile sur $\bC$ en un simple problème de comptage
sur un corps fini. 

On veut démontrer le théorème suivant: 

\begin{theo}[Ax-Grothendieck]\label{theo-Ax-Grothendieck}
Soit $f : \bC^n \rightarrow \bC^n$ une application polynomiale. Si $f$ est
injective, alors elle est bijective. 
\end{theo}

On va suivre le plan suivant :

\begin{enumerate}[label= (\roman*)]
\item On traduit l'énoncé en une formule du premier ordre dans le langage des anneaux.
\item On invoque le principe de Lefschetz, pour démontrer le résultat sur $\bC$,
il suffit de le montrer sur $\Fpalg$ pour tout $p \in \bP$.
\item On se restreint à un sous-corps fini en raisonnant par l'absurde.
\item Le résultat est évident sur un corps fini.

\end{enumerate}

Pour traduire notre énoncé en logique du premier ordre, on va avoir besoin du lemme suivant: 

\begin{lemm} \label{lemm-phrase-corps}
Pour tout couple $(n,d)$ d'entiers positifs, il existe une phrase
$\varphi_{n,d}$ dans le langage des anneaux qui exprime que dans un corps $K$,
toute famille $(f_1, \ldots, f_n)$ de polynômes en $n$ variables de degré
inférieur ou égal à $d$, si l'application associée est injective, alors elle est surjective.
\end{lemm}

\begin{exem}
Construisons la phrase pour l'exemple $n=3$ et $d=2$, la construction générale
est la même.

Un polynôme $P$ en $3$ variables de degré $\le 2$ est une combinaison linéaire des $10$ monômes suivants:
    \[ 1, \quad x_1, x_2, x_3, \quad x_1^2, x_2^2, x_3^2, \quad x_1x_2, x_1x_3, x_2x_3 \]
    Une application polynomiale $f: K^3 \to K^3$ est définie par un triplet de polynômes $(f_1, f_2, f_3)$.
    Chacun nécessitant $10$ coefficients, l'application est donc entièrement déterminée
    par un tuple de $3 \times 10 = 30$ coefficients.
    Considérons 30 variables (au sens logique) que nous noterons globalement par
    le vecteur $\overline{c} = (c_k,d)$. 
    Pour chaque composante $k \in \{1,2,3\}$, on peut écrire formellement le terme polynomial associé :
    \[ f_k(\overline{c}, \overline{x}) = \sum_{\delta \le 2} c_{k,\delta} \cdot \overline{x}^\delta \]

    L'injectivité s'exprime par la formule:
    \[ \Inj(\overline{c}): \forall \overline{x} \forall \overline{y} \left( \left[ \bigwedge_{k=1}^3 f_k(\overline{c}, \overline{x}) = f_k(\overline{c}, \overline{y}) \right] \longrightarrow (\overline{x} = \overline{y}) \right) \]
    La surjectivité s'exprime par la formule:
    \[  \Surj(\overline{c}) :\forall \overline{z} \exists \overline{w} \left( \bigwedge_{k=1}^3 f_k(\overline{c}, \overline{w}) = z_k \right) \]

    Et on a alors:
    \[ \varphi_{3,2} : \forall \overline{c} \left( \Inj(\overline{c}) \longrightarrow
    \Surj(\overline{c}) \right) \]
    
    qui est bien une formule de longueur finie du premier ordre dans le langage des anneaux.
\end{exem}

\begin{proof}
D'après le \cref{lemm-phrase-corps} et par principe de Lefschetz (\cref{coro-Lefschetz}), il suffit de
montrer le résultat
pour $\Fpalg$ pour tout $p \in \bP$. Par le \cref{theo-structure-corps-fini}, toute partie finie de $\Fpalg$ est contenue dans
un sous-corps fini de $\Fpalg$. Soit $f = (f_1, \ldots, f_n) : \Fpalg^n
\rightarrow \Fpalg^n$ ,  une application polynomiale injective. Supposons qu'il
existe $y \in \Fpalg^n$ qui n'est pas dans l'image de $f$. Soit $K$ le sous-corps de $\Fpalg$ qui contient toutes les composantes de $y$ et tous les
coefficients des $f_i$. $f$ induit alors une application $K^n \rightarrow K^n$
qui est injective et non surjective ce qui est absurde car $K$ est fini d'après le
\cref{theo-structure-corps-fini}.
\end{proof}
